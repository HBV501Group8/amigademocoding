Example Vectorballs and Sorting

assembly.asm

In this example, we will create a simple demo of vector balls (bobs) bouncing around the screen.

What are Vectorballs? You define a set of 3D points (for example, forming a shape like a cube, logo, or sphere). Each point is rendered as a filled circle (ball) on screen. Instead of drawing lines (like wireframe), you draw solid discs, giving a more organic and smooth look.

3D coordinates Each ball has (x, y, z) Transformations Rotate / move points in 3D Projection Convert to 2D: screenX = x / z screenY = y / z Size & shading Ball size depends on Z (closer = bigger) Color/brightness also often depends on Z

Here is a sorting roytine for the vector balls. We sort them by their Z value (depth) so that we can draw them in the correct order (back to front) for proper occlusion.

	move.l	#32,d7
	subq.l	#1,d7		; On 1st sort, look at (no. of points-1) after 1st,
				; -1 for dbra
Sort1
	move.l	d7,d6		; Keep d7 the same, use d6 instead
	move.l	a0,a1		; Ditto a0 and a1
	addq.l	#8,a1		; Start searching in entry after current one
	sub.l	a2,a2		; No smaller value found so far
	move.w	4(a0),d0	; d0=value to compare each entry with
.loop	cmp.w	4(a1),d0	; Is this entryd0 
	ble	.ok		; Nope
	move.w	4(a1),d0	; d0=this entry
	move.l	a1,a2		; a2=address of this entry
.ok	addq.l	#8,a1		; Search others for smaller entry
	dbra	d6,.loop
	cmp.l	#0,a2		; Was a smaller value found?
	beq	.ok1		; Nope
	move.l	(a0),d0		; Swap first four bytes
	move.l	(a2),(a0)
	move.l	d0,(a2)+
	move.l	4(a0),d0	; Swap secound four bytes
	move.l	(a2),4(a0)
	move.l	d0,(a2)+
.ok1:
	addq.w	#8,a0		; a0=address of next entry to sort
	dbra	d7,Sort1	; sort other entries
	rts
WS:
	lea.l	WSpace,a0

	move.w	#200,(a0)+	Store x
	move.w	#10,(a0)+	Store y
	rts

	cmp.l	#0,a2		; Was a smaller value found?
	beq	.ok1		; Nope
	move.l	(a0),d0		; Swap first four bytes
	move.l	(a2),(a0)
	move.l	d0,(a2)+
	move.l	4(a0),d0	; Swap secound four bytes
	move.l	(a2),4(a0)
	move.l	d0,(a2)+
.ok1:
	addq.w	#8,a0		; a0=address of next entry to sort
	dbra	d7,Sort1	; sort other entries
	rts
WS:
	lea.l	WSpace,a0

	move.w	#200,(a0)+	Store x
	move.w	#10,(a0)+	Store y
	rts


BlitBalls	
	lea	custom,a6
	lea	wspace,a0		; A0=Address of new points
	lea	ballart,a1
	move.w	#0,d0			; D0=Number of balls to blit
BlitLoop
	moveq.l	#0,d1			; Clear D1
	move.w	(a0),d1			; D1=X pos
*	move.w	#100,d1			; D1=X pos
	ext.l	d1			; Make D1 long
*	add.l	#160,d1			; Add middle of screen
	divu	#16,d1			; Convert
	moveq.l	#0,d3			; Clear D3
	move.w	2(a0),d3		; D3=Y pos
*	move.w	#100,d3		; D3=Y pos
	ext.l	d3			; Make D3 long
*	add.l	#80,d3			; Add Y centre
	mulu	#40*4,d3		; Convert Y
	moveq.l	#0,d4			; Clear D4
	move.w	d1,d4			; D4=X offset
	add.l	d4,d4			; Turn X offset into bytes
	add.l	d4,d3	
	move.l	wscr,a4		; Add X to Y
	add.l	a4,d3		; Add start of screen to X Y
	swap	d1	
	lsl.w	#8,d1			; Put shift in 4 MSB
	lsl.w	#4,d1
	move.w	#$ffa,d2	; D2=Bltcon0 value
	or.w	d1,d2			; OR shift value
.Wait	btst	#14,dmaconr(a6)		; Wait for blitter
	bne	.Wait
	move.l	#Vectorbob,a2	; Source=Vector bob
	move.l	#VectorMask,a3	; Source=Vector bob mask
	move.w	d0,d6
*	addq	#1,d6
	add.w d6,d6
	clr.l	d5
	move.w	(a1,d6.w),d5
row3
	cmp.w	#40,d5
	blt	row2
	clr.l	d6
	sub.w	#40,d5
	add.w	d5,d5
	add.w	d5,d5
	move.w	#42*32*4,d6
	add.w	d5,d6
	MOVE.W	#34,blitmodscreen
	move.w	#36,blitmodvbob
	move.w	#(64*31*4)+3,blitsize
	bra	rowon
row2	

	cmp.w	#20,d5
	blt	row1

	clr.l	d6
	move.w	d5,d6
	sub.w	#20,d6
	add.w	#42*16*4,d6
	
	MOVE.W	#36,blitmodscreen
	move.w	#38,blitmodvbob
	move.w	#(64*16*4)+2,blitsize
	bra	rowon
row1
	mulu.w	#2,d5
	move.w	d5,d6
	MOVE.W	#36,blitmodscreen
	move.w	#38,blitmodvbob
	move.w	#(64*16*4)+2,blitsize

rowon	

	add.w	d6,a2
	add.w	d6,a3

	move.l	a2,bltapth(a6)	; Source=Vector bob
	move.l	a3,bltbpth(a6)	; Source=Vector bob mask
	
	move.l	d3,bltcpth(a6)		; Source=Screen
	move.l	d3,bltdpth(a6)	
*	rts	; Destination=Screen
	move.w	blitmodscreen,bltcmod(a6)		; 2-4
	move.w	blitmodscreen,bltdmod(a6)		; 4-4
	move.w	blitmodvbob,bltamod(a6)
	move.w	blitmodvbob-2,bltbmod(a6)
	move.w	#$ffff,bltafwm(a6)	; No FWM
	move.w	#$0000,bltalwm(a6)	; Full LWM
	move.w	d2,bltcon0(a6)
	move.w	d1,bltcon1(a6)
	move.w	blitsize,bltsize(a6) ; 16*32

	addq.w	#4,a0
	ADDq	#1,d0			; Point to next balls cords
	CMP.W	#31,d0
	bne	blitloop		; Blit all balls
	rts
