Using only projection we can't see the object from all angles and that's where rotation comes in.

Coordinate system (right-handed) Let’s assume the standard right-handed system used in most math/graphics: X axis → left to right Y axis → up and down Z axis → out of / into the screen

FORMULA FOR ROTATIONS IN THREE DIMENSION WITH THREE ANGLES (x,y,z is the original (x,y,z) coordinate. c1=cos(angle1), s1=sin(angle1), c2=cos(angle2) and so on...) If you want to rotate a lot of coordinates with the same angles you first calculate these values: 
  
                       xx=c2*c1
                       xy=c2*s1
                       xz=s2
                       yx=c3*s1+s3*s2*c1
                       yy=-c3*c1+s3*s2*s1
                       yz=-s3*c2
                       zx=s3*s1-c3*s2*c1;s2*c1+c3*s1
                       zy=-s3*c1-c3*s2*s1;c3*c1-s2*s1
                       zz=c3*c2

Then, for each coordinate, you use the following multiplication to get the rotated coordinates: 

                       x''=xx * x + xy * y + xz * z
                       y''=yx * x + yy * y + yz * z
                       z''=zx * x + zy * y + zz * z

So, you only have to calculate the constants once for every new angle, and THEN you use nine multiplications for every point you wish to rotate to get the new set of points.

Here is my rotation routine based on this formula it takes in sin and cos tables and 3d coords for input and rotcoords for output

	lea 	sin,a0

	and.w	#$3ff,d0
	move.w	d0,thetay
	move.w	(a0,d0),siny	sin y
	move.w	(a1,d0),cosy	cos y
	move.w	thetaz,d0
	add.w	#4,d0
*	add.w	d0,d0
	and 	#$3ff,d0
	move.w	d0,thetaz
	move.w	(a0,d0),sinz	sin z
	move.w	(a1,d0),cosz	cos z
	moveq	#14,d6		
	move.w	#starsnr-1,d7	

rloop
	move.w	(a2)+,d0
	move.w	(a2)+,d1
	move.w	(a2)+,d2

	move.w	d1,d4
	move.w	d2,d5

* Rotation of x axis		   *	
************************************

	muls	cosx,d1 	y * cos(thetax)
	muls	sinx,d2		z * sin(Thetax)
	sub.l	d2,d1		y * cos(thetax) - z * sin(Thetax) 
	asr.l	d6,d1
*	swap 	d1
	move.w	d1,temp1		new y

*	move.w	d3,d0
	move.w	d4,d1
	move.w	d5,d2


	muls	cosx,d2 	z * cos(thetax)
	muls	sinx,d1		y * sin(Thetax)
	add.l	d1,d2		y * cos(thetax) + z * sin(Thetax) 
	asr.l	d6,d2
*	swap	d2
	move.w	d2,temp2		new z
	move.w	temp1,desty
	move.w	temp2,destz

	move.w	d3,d0	
	move.w	destz,d2

************************************
* Rotation of y axis		   *	
************************************

	muls	cosy,d2		z * cos(y)
	muls	siny,d0		x * sin(y)
	sub.l	d0,d2		x * sin(y) - z * cos(y)
	asr.l	d6,d2
*	swap	d2
	move.w	d2,temp1	new z

	move.w	d3,d0
	move.w	destz,d2

	muls	cosy,d0		x * cos(y)
	muls	siny,d2		z * sin(y)
	add.l	d2,d0		x * cos(y) + z * sin(y)
	asr.l	d6,d0
*	swap	d0
	move.w	d0,temp2		new x	

	move.w	temp1,destz
	move.w	temp2,destx

	move.w	destx,d0
	move.w	desty,d1

************************************
* Rotation of z axis		   *	
************************************

	muls	cosz,d0		x * cos(z)
	muls	sinz,d1		y * sin(z)
	sub.l	d1,d0		x * cos(z) - y * sin(z)
	asr.l	d6,d0
*	swap	d0
	move.w	d0,temp1

	move.w	destx,d0
	move.w	desty,d1
			 	
	muls	cosz,d1
	muls	sinz,d0
	add.l	d0,d1
	asr.l	d6,d1
*	swap	d1
	move.w	d1,temp2

	move.w	temp1,destx
	move.w	temp2,desty

	move.w	destx,d0
	move.w	desty,d1
	move.w	destz,d2

	move.w	d0,(a3)+
	move.w	d1,(a3)+
	move.w	d2,(a3)+


	dbra	d7,rloop	
	rts

; Now blit the balls to the screen
BlitBalls	
	lea	custom,a6
	lea	wspace,a0		; A0=Address of new points
	lea	ballart,a1
	move.w	#0,d0			; D0=Number of balls to blit
BlitLoop
	moveq.l	#0,d1			; Clear D1
	move.w	(a0),d1			; D1=X pos
*	move.w	#100,d1			; D1=X pos
	ext.l	d1			; Make D1 long
*	add.l	#160,d1			; Add middle of screen
	divu	#16,d1			; Convert
	moveq.l	#0,d3			; Clear D3
	move.w	2(a0),d3		; D3=Y pos
*	move.w	#100,d3		; D3=Y pos
	ext.l	d3			; Make D3 long
*	add.l	#80,d3			; Add Y centre
	mulu	#40*4,d3		; Convert Y
	moveq.l	#0,d4			; Clear D4
	move.w	d1,d4			; D4=X offset
	add.l	d4,d4			; Turn X offset into bytes
	add.l	d4,d3	
	move.l	wscr,a4		; Add X to Y
	add.l	a4,d3		; Add start of screen to X Y
	swap	d1	
	lsl.w	#8,d1			; Put shift in 4 MSB
	lsl.w	#4,d1
	move.w	#$ffa,d2	; D2=Bltcon0 value
	or.w	d1,d2			; OR shift value
.Wait	btst	#14,dmaconr(a6)		; Wait for blitter
	bne	.Wait
	move.l	#Vectorbob,a2	; Source=Vector bob
	move.l	#VectorMask,a3	; Source=Vector bob mask
	move.w	d0,d6
*	addq	#1,d6
	add.w d6,d6
	clr.l	d5
	move.w	(a1,d6.w),d5
row3
	cmp.w	#40,d5
	blt	row2
	clr.l	d6
	sub.w	#40,d5
	add.w	d5,d5
	add.w	d5,d5
	move.w	#42*32*4,d6
	add.w	d5,d6
	MOVE.W	#34,blitmodscreen
	move.w	#36,blitmodvbob
	move.w	#(64*31*4)+3,blitsize
	bra	rowon
row2	

	cmp.w	#20,d5
	blt	row1

	clr.l	d6
	move.w	d5,d6
	sub.w	#20,d6
	add.w	#42*16*4,d6
	
	MOVE.W	#36,blitmodscreen
	move.w	#38,blitmodvbob
	move.w	#(64*16*4)+2,blitsize
	bra	rowon
row1
	mulu.w	#2,d5
	move.w	d5,d6
	MOVE.W	#36,blitmodscreen
	move.w	#38,blitmodvbob
	move.w	#(64*16*4)+2,blitsize

rowon	

	add.w	d6,a2
	add.w	d6,a3

	move.l	a2,bltapth(a6)	; Source=Vector bob
	move.l	a3,bltbpth(a6)	; Source=Vector bob mask
	
	move.l	d3,bltcpth(a6)		; Source=Screen
	move.l	d3,bltdpth(a6)	
	move.w	blitmodscreen,bltcmod(a6)		; 2-4
	move.w	blitmodscreen,bltdmod(a6)		; 4-4
	move.w	blitmodvbob,bltamod(a6)
	move.w	blitmodvbob-2,bltbmod(a6)
	move.w	#$ffff,bltafwm(a6)	; No FWM
	move.w	#$0000,bltalwm(a6)	; Full LWM
	move.w	d2,bltcon0(a6)
	move.w	d1,bltcon1(a6)
	move.w	blitsize,bltsize(a6) ; 16*32

	addq.w	#4,a0
	ADDq	#1,d0			; Point to next balls cords
	CMP.W	#31,d0
	bne	blitloop		; Blit all balls
	rts
