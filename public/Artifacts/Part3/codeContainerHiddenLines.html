How can we change from a wireframe cube to a solid cube? We need to check if a face (plane) is not visible and avoid drawing it, since it makes no sense to render a plane that is not visible.

How can we achieve this? This is where the dot product comes in.

In 3D graphics, backface culling is a process used to avoid drawing the faces of 3D objects that are facing away from the camera. Every polygon (usually a triangle) has a surface normal vector — the direction the face is pointing. If a face is pointing away from the camera, we don’t need to render it because it is invisible from the current view. That is what backface culling does: it skips faces that face away.

The Math regarding dot product

So we need to find the Z component of a plane’s normal to determine if it is visible. A plane consists of at least 3 points, but that is enough to calculate the normal.

normal_Z:
	movem.l	d1-d6,-(sp)

	move.w	0(a2),d6		; coord 1
	muls	#6,d6		; *4
	move.w	(a3,d6.w),d0	; X1
	move.w	2(a3,d6.w),d1	; Y1

	move.w	4(a2),d6	; coord 2
	muls	#6,d6		; *4
	move.w	(a3,d6.w),d2	; X2
	move.w	2(a3,d6.w),d3	; Y2

	move.w	2(a2),d6	; coord 3
	muls	#6,d6		; *4
	move.w	(a3,d6.w),d4	; X3
	move.w	2(a3,d6.w),d5	; Y3

	sub.l	d4,d0		; (X1-X3)=Vx1
	sub.l	d5,d1		; (Y1-Y3)=Vy1
	sub.l	d4,d2		; (X2-X3)=Vx2
	sub.l	d5,d3		; (Y2-Y3)=Vy2

	muls.w	d3,d0		; Vx1*Vy2
	muls.w	d2,d1		; Vy1*Vx2

	sub.l	d1,d0		; (Vx1*Vx2)-(Vy1*Vy2) If d0 is negative then plane is not visible
	movem.l	(sp)+,d1-d6
	rts

normal_Z:
	movem.l	d1-d6,-(sp)

	move.w	0(a2),d6		; coord 1
	muls	#6,d6		; *4
	move.w	(a3,d6.w),d0	; X1
	move.w	2(a3,d6.w),d1	; Y1

	move.w	4(a2),d6	; coord 2
	muls	#6,d6		; *4
	move.w	(a3,d6.w),d2	; X2
	move.w	2(a3,d6.w),d3	; Y2

	move.w	2(a2),d6	; coord 3
	muls	#6,d6		; *4
	move.w	(a3,d6.w),d4	; X3
	move.w	2(a3,d6.w),d5	; Y3

	sub.l	d4,d0		; (X1-X3)=Vx1
	sub.l	d5,d1		; (Y1-Y3)=Vy1
	sub.l	d4,d2		; (X2-X3)=Vx2
	sub.l	d5,d3		; (Y2-Y3)=Vy2

	muls.w	d3,d0		; Vx1*Vy2
	muls.w	d2,d1		; Vy1*Vx2

	sub.l	d1,d0		; (Vx1*Vx2)-(Vy1*Vy2)

*	lsr.w	#8,d2
*	lsr.w	#3,d2
*	and.w	#$f,d2
*	move.w	d2,color
	movem.l	(sp)+,d1-d6
	rts

*****************************
* This is line draw routine *
* d0 = x0  :  d1 = y0	    *
* d2 = x1  :  d3 = y1	    *
* d4 = screenwidth 	    *
* a0 = pointer to bitmap    *
*****************************
DrawLine
	movem.l	d0-d7/a0-a6,-(sp)	Save regs
	
	bsr	WBlit
	move.w	d4,bltcmod(a6)		C mod = bitplane width
	move.w	d4,bltdmod(a6)		D mod = bitplane width

	mulu.w	d1,d4
	move.w	d0,d5		d5=x1
	lsr.w	#3,d5		d5=x1/8
	add.w	d5,d4		add d5 to d4
	add.l	d4,a0		add bitplane-adress to a0

	lea.l	Octant_table,a4
* Calculate octants & delta

	moveq.l	#0,d5			Clear work register
	sub.w	d1,d3			y2-y1  DeltaY from d3
	roxl.b	#1,d5			Scroll neg-sign in deltY
	tst.w	d3			Restore n-flag ?
	bge.s	y2gy1			Jump if deltaY is positive
	neg.w	d3			Else invert deltaY
y2gy1	sub.w	d0,d2			x2-x1  DeltaX
	roxl.b	#1,d5			Scroll neg-sign in deltx
	tst.w	d2			Restore n-flag ?
	bge.s	x2gx1			Jump if deltax is positive
	neg.w	d2			Else invert deltay
x2gx1	move.w	d3,d1			D1 = deltaY
	sub.w	d2,d1			d1 = deltaY-deltaX
	bge.s	dygdx			Jump if deltay>deltax
	exg.l	d2,d3			Smaller delta goto d2
dygdx	roxl.b	#1,d5			d5=number of actual octant
	move.b	(a4,d5),d5		d5=octant-code
	add.w	d2,d2			d2=smallest data*2
	move.w	d2,bltbmod(a6)		B mod = 2 * smaller delta

	and.w	#$000f,d0		Get 4 bits of x1
	ror.w	#4,d0			Set shift-value
	or.w	#$0bfa,d0	$0bfa = Or with USEA+USEC+USED, D=A+X
*				$0b5a =	Eor with USEA+USEC+USED,D=Ac+aC
	move.l	d3,d1			Set up size word
	lsl.w	#6,d3			Shift six left (blitter height)
	add.w	#$42,d3			And add one to height, 2 to width

	sub.w	d1,d2			2* Smaller delta - Larger delta
	bge.s	signnl			Jump if 2*SmallerDelta > GreaterDelta
	or.b	#SignFlag,d5		Set sign bit in bltcon1
signnl	move.w	d2,bltaptl(a6)		2* Smaller delta - Larger delta
	sub.w	d1,d2			2* Smaller delta - 2* Larger delta
	move.w	d2,bltamod(a6)		To bltamod

	move.w	d0,bltcon0(a6)		Set shift and minterms
	move.w	d5,bltcon1(a6)		Set ocant number
	move.w	#$8000,bltadat(a6)	A data = 0x8000
	moveq.l	#-1,d5			Set mask all to ones
	move.l	d5,bltafwm(a6)		We can hit both masks at once
	move.l	a0,bltcpth(a6)		Pointer to first pixel to set
	move.l	a0,bltdpth(a6)		Pointer to first pixel to set
	move.w	d3,bltsize(a6)		Start line-draw
	movem.l	(sp)+,d0-d7/a0-a6

	rts				And return, blit still in progress

octant_table
	dc.b	Octant2*4+LineMode	y1dy
	dc.b	Octant3*4+LineMode	y1x2, dxx2, dx>dy
	dc.b	Octant7*4+LineMode	y1>y2, x1y2, x1dy
	dc.b	Octant6*4+LineMode	y1>y2, x1>x2, dxy2, x1>x2, dx>dy
