Example: Plasma done by using Copper waits

assembly.asm

Copper plasma is created by reserving space in the Copper list for 40 MOVE instructions per vertical line, giving a vertical resolution of 8 pixels for the plasma effect.

This layout defines the area in which the plasma will be rendered across the desired screen height.

Once the Copper list has been prepared, each frame uses the Blitter to copy and update the data. A copy operation is performed vertically, one word at a time, across the full height of the effect.

This process continues across the full width of the plasma until the entire screen area is updated.

In this setup, the color table acts as the source, while the Copper list serves as the destination (updated one word at a time, corresponding to 16 pixels).

Below is an example of Copper plasma:

;****************************************************************************
; This routine initializes the copperlist that generates the plasma. It arranges the
; WAIT statements and the first half of the COPPERMOVE statements. At the end of the plasma
; line, a final COPPERMOVE is inserted that loads the
; black color into COLOR00.
;************************************************************************

InitPlasma:
	lea	Plasma,a0	; indirizzo plasma
	move.l	#$403bFFFE,d0	
	move.w	#$180,d1	
	move	#Plasma_Height-1,d3	
InitLoop1:
	move.l	d0,(a0)+		; scrive la WAIT
	add.l	#$01000000,d0		
	moveq	#Plasma_Width,d2		
InitLoop2:
	move.w	d1,(a0)+		
				; "copper move"
	addq.l	#2,a0			
	dbra	d2,InitLoop2
	dbra	d3,InitLoop1
	rts

************************************************************
; This routine creates the plasma. It makes a loop of blitts, each
; of which writes a "column" of the plasma, that is, it writes the colors in the
; COPPERMOVES placed in a column.
; The colors written in each column are read from a table, starting from
; an address that varies from one column to another based on offsets
; read from another table. Furthermore, between one frame and another the offsets
; vary, creating the movement effect.
;****************************************************************************

DoPlasma:
	lea	Color,a0		
	lea	Sin,a3		
	lea	Plasma+6,a1		
	move.w	Sinadd(pc),d0		;
	sub.w	#NewFrame,d0		
	and.w	#$03FF,d0	
	move.w	d0,Sinadd		;
	move.w	#Plasma_Height*64+1,d3	; dimensione
	moveq	#Plasma_Width-1,d2	
PlasmaLoop:				; inizio loop blittate
	move.w	(a3,d0.w),d1		; legge offset dalla tabella
	lea	(a0,d1.w),a2	

	Btst	#6,2(a6)
WaitBlit:
	Btst	#6,2(a6)		
	bne.s	WaitBlit
	move.l	#$09f00000,BLTTCON0(a6)	
	move.l  #-1,BLTAFWM(a6)		; BLTAFWM/BLTALWM
	move.w	#0,bltamod(a6)
	move.w	#bytes_per_line-2,bltdmod(a6)
	move.l	a2,BLTAPT(a6)		; BLTAPT 
	move.l	a1,BLTDPT(a6)		; BLTDPT 
	move.w	d3,$BLTSIZE(a6)		; BLTSIZE

	addq.w	#4,a1			
	add.w	#NewLine,d0		
	and.w	#$03FF,d0		

	dbra	d2,PlasmaLoop
	rts




Sinadd:	dc.w	0

; Input: D0 = number to convert
; Output: buffer contains ASCII string (null-terminated)
; Uses: D0, D1, D2, A0

        LEA     buffer,A0        ; pointer to output buffer
        MOVE.L  D0,D1            ; copy value to D1 (so we can modify it)
        LEA     endbuf,A0        ; start filling digits backwards
        MOVE.B  #0,-(A0)         ; null terminator

convert_loop:
        DIVU    #10,D1           ; D1 / 10 -> quotient in upper word, remainder in lower word
        SWAP    D1               ; remainder now in lower word
        ADDI.B  #'0',D1          ; convert remainder to ASCII
        MOVE.B  D1,-(A0)         ; store digit
        SWAP    D1               ; swap back so quotient is in lower word
        MOVE.W  D1,D1            ; clear upper word (since DIVU result is in lower word only)
        TST.W   D1               ; check if quotient == 0
        BNE     convert_loop
		rts
