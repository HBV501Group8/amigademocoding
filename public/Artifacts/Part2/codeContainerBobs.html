Example: M68000 Assembly Blitting objects

assembly.asm

Sine bobs are created using sine and cosine functions to generate smooth circular or wave-like motion. These equations can become quite complex, and the resulting X and Y coordinates are typically stored in a buffer.

A separate routine then reads these coordinates and renders so-called BOBs (Blitter Objects) using the Blitter.

This is the basic principle behind the effect.

To blit a bob, the X and Y coordinates must be converted into a format that the Blitter can use for screen addressing and pixel placement.

Below is an example:

	move.w	intenar(a6),SysIntenar		Get system inenta
	move.w	intreqr(a6),SysIntreqr		Get system intreq
	move.w	#$7fff,intreq(a6)		Clear system enable

	or.w	#$8000,SysIntreqr		Set intreq set bit
	move.w	SysIntreqr(pc),intreq(a6)	Restore it
	move.w	SysIntenar(pc),intena(a6)	Restore it

***********************************************************
* This kills the system DMA and then sets my dma and when *
* all that it is done it is capable of restoring it again *
***********************************************************
SetDMA	move.w	DMAconr(a6),d0		Fetch system dma status
	move.w	d0,SystemDMA		Save system dma
	move.w	#$7fff,DMAcon(a6)	Clear all dma
	move.w	#$83e0,DMAcon(a6)	Set COPEN+BPLEN+DMAEN+BLTEN
	rts

RestoreDMA
	move.w	#$7fff,DMAcon(a6)	Clear all dma
	move.w	SystemDMA(pc),d0	Get old system dma
	or.w	#$8000,d0		And set
	move.w	d0,DMAcon(a6)		And now start it
	rts

	move.b	#%00000000,CIAASDR	Clear serial data register
	move.b	#%10000000,CIAACRA	Set control register A

	LEA BOB,A1			    ;

bobloop:	
*	MOVE.W (A1),D1			;POSITION X  D1
*	MOVE.W 2(A1),D3			;POSITION Y  D3
	MOVE.W (a0),D1			;POSITION X  D1
	MOVE.W 2(a0),D3			;POSITION Y  D3
	MULU #ScreenDepth,D3	;MULTIPLY BY ScreenDepth
	MOVE.W	D1,D2			;MOVE X  D2
	AND.W	#$FFF0,D1		;Mask
	LSR.W	#3,D1			;DIVIDE BY 8
	ADD.W D3,D1		    	;ADD TO Y POS
					;
	
	AND.W	#$000F,D2		;MASK 12 BITS
	MULU	#$1000,D2		;MULTIPLY BY $1000
					        
	MOVE.W D1,4(A1)			
	MOVE.W D2,6(A1)			
	
	move.l	Wscr,A4			;WSCR IN A4

	ADD.W 4(a1),A4			

	LEA SPR,A3			;GET BOB IN A3
	LEA MSPR,A5			;AND THE MASK IN A5

	jSR 	WAIT_BLITTER		
	MOVE.L A5,BLTAPTH(a6)		
	MOVE.L A3,BLTBPTH(a6)		
	MOVE.L A4,BLTCPTH(a6)		
	MOVE.L A4,BLTDPTH(a6)		

	MOVE.W #34,BLTAMOD(a6)		;40 - 6 MODULO SOURCE A
	MOVE.W #34,BLTBMOD(a6)		;40 - 6 MODULO SOURCE B
	MOVE.W #34,BLTCMOD(a6)		;40 - 6 MODULO SOURCE C
	MOVE.W #34,BLTDMOD(a6)		;40 - 6 ODULO DESTINATION D
	
	MOVE.L #$FFFF0000,d2		;MASK UPPER HALF 
*	MOVE.W 0,D3			
	MOVE.W 6(A1),D3		
	ROR.L D3,D2			
	MOVE.L D2,bltafwm(A6) 		;SET MASK
	
	MOVE.W 6(A1),BLTCON1(a6)	
	MOVE.W #$0FCA,D4		;$F = 3 SOURCE + DESTINATION  LOGIC
					        ;$CA = MINTERM 
	ADD.W 6(A1),D4			
	MOVE.W D4,BLTCON0(a6)		;SET BLTCON0
	
	MOVE.W #(32*ScreenDepth*64)+(6/2),BLTSIZE(a6)	;CHARGE TAILLE POUR DEMARRER BLITTER
	add.w	#4,a0
	dbf	d7,bobloop
	
	RTS
